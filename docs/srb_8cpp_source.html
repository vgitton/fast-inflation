13inf::TargetDistr::ConstPtr user::get_noisy_srb(Num visibility, Num visibility_denom) {
14 Num const max_noise_level = visibility_denom;
15
16 Num const noise_level = max_noise_level - visibility;
17
18 ASSERT_LT(noise_level, max_noise_level + 1)
19 ASSERT_LT(-1, noise_level)
20
21 inf::Network::ConstPtr const network = user::get_srb_network();
22
23 inf::EventTensor d(network->get_n_parties(), network->get_n_outcomes());
24 Num const denom = max_noise_level * 8;
25 d.set_denom(denom);
26
27 for (inf::Event const &e : network->get_event_range()) {
28 d.get_num(e) = noise_level;
29 if (e[0] == e[1] and e[1] == e[2])
30 d.get_num(e) += (max_noise_level - noise_level) * 4;
31 }
32
33 return std::make_shared<inf::TargetDistr>(
34 "Noisy shared random bit with visibility " + inf::VisProblem::visibility_to_str(visibility, visibility_denom),
35 "SRB", network, d);
36}

132void user::srb_dual_vector_io::run() {
133 (*get_feas_options())
134 .set(inf::Inflation::Size{2, 2, 2})
135 .set(inf::ConstraintSet::Description{
136 {"A00,B00,C00", "A11,B11,C11", ""},
137 {"A00", "A11,B10,B11,C01,C11"},
138 });
139
140 Num const vis_denom = 100000;
141 Num const vis_certificate = 41422;
142 std::string const filename = "data/srb_222_certificate";
143 std::string const metadata = "A nonlocality certificate for the Shared Random Bit with visibility " +
144 inf::VisProblem::visibility_to_str(vis_certificate, vis_denom) +
145 " obtained with inflation size 2x2x2 and all LPI constraints";
146
147 {
148 inf::FeasProblem feas_pb(user::get_noisy_srb(vis_certificate, vis_denom), get_feas_options());
149 inf::FeasProblem::Status const status = feas_pb.get_feasibility();
150 HARD_ASSERT_TRUE(status == inf::FeasProblem::Status::nonlocal)
151
152 feas_pb.write_dual_vector_to_file(filename, metadata);
153 }
154
155 {
156 inf::FeasProblem feas_pb(user::get_noisy_srb(0, vis_denom), get_feas_options());
157
158 // Want the new vis to be even to ensure that the bounds for the stored quovec are respected.
159 // If one really wanted to test the stored nonlocality certificate for odd visibilities, we would need
160 // to implement a mechanism to rescale the quovec components to lower values to ensure that the overflow
161 // mechanism is respected, but this is not really needed for our purposes.
162 for (Num const vis_offset : util::Range(-5, 5)) {
163 Num const vis = vis_certificate + 2 * vis_offset;
164 feas_pb.update_target_distribution(user::get_noisy_srb(vis, vis_denom), inf::FeasProblem::RetainEvents::no);
165 // Need to read the dual vector after updating the target distribution, since updating the taret distribution
166 // resets the dual vector to all zeros
167 feas_pb.read_dual_vector_from_file(filename, metadata);
168
169 inf::Optimizer::Solution const sol = feas_pb.minimize_dual_vector();
170 bool const nonlocal = sol.get_inflation_event_score() > 0;
171
172 util::logger << "Visibility " << inf::VisProblem::visibility_to_str(vis, vis_denom)
173 << " is ";
174 if (nonlocal) {
175 util::logger << "nonlocal";
176 HARD_ASSERT_LTE(vis_certificate, vis)
177 } else {
178 util::logger << "inconclusive";
179 HARD_ASSERT_LT(vis, vis_certificate)
180 }
181 util::logger << " based on the certificate" << util::cr;
182 }
183 }
184
185 util::logger << util::cr;
