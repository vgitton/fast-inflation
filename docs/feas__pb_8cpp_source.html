78 std::cout << std::flush;

81

109 // Using \sqrt{2} as a ratio, since this gives more or less double the time for each print statement
110 if (static_cast<double>(m_n_iterations) >= 1.4142 * static_cast<double>(last_printed_iteration)) {
111 util::logger << "Iteration " << m_n_iterations << ", "

120
121 util::logger << util::flush;
122
123 last_printed_iteration = m_n_iterations;
124 }
125 }
126
127 if (not fw_sol.valid) {
128 util::logger << "Iteration " << m_n_iterations << ", "
129 << "s = " << fw_sol.s << ", ";
130 m_optimizer->log_status();
131 log_status_bar();
132
133 status = inf::FeasProblem::Status::inconclusive;
134
135 if (not new_display_style)
136 util::logger << " -> BREAK";
137
138 break;
139 }
140
141 // m_optimizer holds a reference to the constraint set, and will minimize
142 // the potential separating hyperplane given by the dual vector
143 m_constraint_set->set_dual_vector_from_quovec(round_dual_vector(fw_sol.vec));
144
145 if (not new_display_style)
146 util::logger << util::cr << "Optimizing...";
147
148 inf::Optimizer::Solution const sol = minimize_dual_vector();
149
150 if (not new_display_style)
151 util::logger << util::cr << sol;
152
153 if (sol.get_inflation_event_score() > 0) {
154 if (m_n_iterations > 1) {
155 util::logger << "Iteration " << m_n_iterations << ", "
156 << "s = " << fw_sol.s << ", ";
157 m_optimizer->log_status();
158 log_status_bar();
159 }
160
161 util::logger << sol << util::cr;
162
163 status = inf::FeasProblem::Status::nonlocal;
164
165 if (not new_display_style)
166 util::logger << " -> BREAK";
167
168 break;
169 }
170
171 if (not new_display_style)
172 util::logger << util::cr;
173
174 // The distribution might be feasible...
175 memorize_event(sol.get_inflation_event());
176 }
177
178 LOG_END_SECTION
179 util::logger.cancel_go_back_to_mark();
180 this->log_feasibility(status);
181 util::logger << util::flush;
182 return status;

185void inf::FeasProblem::update_target_distribution(inf::TargetDistr::ConstPtr const &d,
186 inf::FeasProblem::RetainEvents retain_events) {
187 m_distribution = d;
188 m_n_iterations = 0;
189
190 try {
191 m_constraint_set->set_target_distribution(*d);
192 } catch (inf::Symmetry::SymmetriesHaveChanged const &e) {
193 THROW_ERROR(util::str("the target distribution's symmetries have changed, ") +
194 "but re-creating the inf::FeasProblem members is currently unimplemented," +
195 "please re-create an inf::FeasProblem entirely.")
196 }
197
198 std::set<inf::Event> prime_events;
199 if (retain_events == inf::FeasProblem::RetainEvents::yes) {
200 // Need to get the prime events before resetting!
201 prime_events = m_frank_wolfe->get_stored_events();
202 }
203
204 m_frank_wolfe->reset();
205
206 if (retain_events == inf::FeasProblem::RetainEvents::yes) {
207 for (inf::Event const &event : prime_events)
208 memorize_event(event);
209 } else if (retain_events == inf::FeasProblem::RetainEvents::no) {
210 init_frank_wolfe();
211 } else
212 THROW_ERROR("switch")
213}

270inf::Quovec inf::FeasProblem::round_dual_vector(std::vector<double> const &dual_vector_double) const {
271 ASSERT_EQUAL(dual_vector_double.size(), m_constraint_set->get_quovec_size())
272
273 inf::Quovec dual_vector_rounded(m_constraint_set->get_quovec_size());
274
275 double minimum_fw_sol_val = std::numeric_limits<double>::max();
276 double maximum_fw_sol_val = std::numeric_limits<double>::min();
277
278 for (double component : dual_vector_double) {
279 if (component < minimum_fw_sol_val)
280 minimum_fw_sol_val = component;
281 if (component > maximum_fw_sol_val)
282 maximum_fw_sol_val = component;
283 }
284
285 minimum_fw_sol_val = std::abs(minimum_fw_sol_val);
286 maximum_fw_sol_val = std::abs(maximum_fw_sol_val);
287
288 double const largest_value = std::max(minimum_fw_sol_val, maximum_fw_sol_val);
289
290 double const scale_factor = 0.95 * static_cast<double>(m_constraint_set->get_max_dual_vector_component()) / largest_value;
291
292 for (Index i : util::Range(m_constraint_set->get_quovec_size())) {
293 dual_vector_rounded[i] = static_cast<Num>(scale_factor * dual_vector_double[i]);
294 }
295
296 util::simplify_by_gcd(dual_vector_rounded);
297
298 return dual_vector_rounded;
299}
