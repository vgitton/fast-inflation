74bool inf::TargetDistr::is_probability_distribution(inf::EventTensor const &event_tensor) {
75 Num norm = 0;
76
77 for (Index hash : event_tensor.get_hash_range()) {
78 Num const the_numerator = event_tensor.get_num(hash);
79
80 if (the_numerator < 0)
81 return false;
82
83 norm += the_numerator;
84 }
85
86 return norm == event_tensor.get_denom();

117inf::EventTensor::UniqueConstPtr inf::TargetDistr::compute_marginal(inf::TargetDistr::MarginalName const &name) const {
118 // We don't want to recompute the full distribution as a marginal,
119 // this is accounted for already in inf::TargetDistr::get_marginal().
120 ASSERT_TRUE(name != this->get_network()->get_party_names())
121
122 // Find the party indices that the name of the marginal refers to
123 std::vector<Index> party_indices;
124 {
125 std::map<std::string, Index> const &party_name_map = this->get_network()->get_party_name_map();
126
127 bool first_party = true;
128 Index previous_party_i = 0;
129
130 for (std::string const &party : name) {
131 HARD_ASSERT_MAP_CONTAINS(party_name_map, party)
132 Index party_i = party_name_map.at(party);
133 if (not first_party) {
134 HARD_ASSERT_LT(previous_party_i, party_i)
135 if (previous_party_i >= party_i)
136 THROW_ERROR("The marginal " + util::str(name) + " is not ordered correctly")
137 } else {
138 first_party = false;
139 }
140
141 party_indices.push_back(party_i);
142
143 previous_party_i = party_i;
144 }
145 }
146
147 Index const marg_n_parties = name.size();
148 inf::EventTensor::UniquePtr marginal = std::make_unique<inf::EventTensor>(marg_n_parties, this->get_network()->get_n_outcomes());
149 marginal->set_denom(m_event_tensor.get_denom());
150 // Initialize the marginal
151 {
152 inf::Event marg_e(marg_n_parties, 0);
153
154 for (inf::Event const &full_e : this->get_event_range()) {
155 for (Index marg_party_i : util::Range(marg_n_parties)) {
156 marg_e[marg_party_i] = full_e[party_indices[marg_party_i]];
157 }
158
159 marginal->get_num(marg_e) += m_event_tensor.get_num(full_e);
160 }
161 }
162 marginal->simplify();
163
164 ASSERT_TRUE(is_probability_distribution(*marginal))
165
166 return marginal;
167}
