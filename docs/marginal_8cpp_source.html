49void inf::Marginal::Evaluator::set_outcome(Index const inflation_party, inf::Outcome const outcome) {
50 ASSERT_LT(outcome, m_n_outcomes)
51 ASSERT_LT(inflation_party, m_inflation_event.size())
52
53 // Don't do anything if the outcome is already set
54 if (m_is_scalar_marginal or outcome == m_inflation_event[inflation_party])
55 return;
56
57 for (inf::Marginal::Evaluator::UpdateRule const &update_rule : m_party_to_update_rules[inflation_party]) {
58 inf::Outcome const new_outcome = update_rule.inverse_outcome_sym[outcome];
59 inf::Outcome const old_outcome = update_rule.inverse_outcome_sym[m_inflation_event[inflation_party]];
60
61 m_marg_event_hashes[update_rule.marg_perm_index] += (new_outcome - old_outcome) * update_rule.party_weight;
62 }
63
64 m_inflation_event[inflation_party] = outcome;
65}

202void inf::Marginal::init_marginal_symmetries(inf::Symmetry::Group const &constraint_group) {
203 ASSERT_EQUAL(m_marginal_symmetries.size(), 0)
204 ASSERT_LT(0, constraint_group.size())
205
206 if (m_marginal_parties.size() == 0)
207 return;
208
209 // These two are used to do the down-conversion
210 std::map<Index, Index> inf_to_marg_index;
211 std::map<Index, Index> marg_to_inf_index;
212 for (Index const marg_party_index : util::Range(get_n_parties())) {
213 Index const inf_party_index = m_marginal_parties[marg_party_index];
214
215 inf_to_marg_index[inf_party_index] = marg_party_index;
216 marg_to_inf_index[marg_party_index] = inf_party_index;
217 }
218
219 for (inf::Symmetry const &inf_sym : constraint_group) {
220
221 inf::PartySym::Bare const &inf_party_sym = inf_sym.get_party_sym().get_bare_sym();
222 inf::PartySym::Bare marg_party_sym(get_n_parties(), 0);
223
224 for (Index const marg_party_index : util::Range(get_n_parties())) {
225 marg_party_sym[marg_party_index] =
226 // inf to marg
227 inf_to_marg_index.at(
228 // inf symmetry
229 inf_party_sym[
230 // marg to inf
231 marg_to_inf_index.at(marg_party_index)]);
232 }
233
234 // Here expect that some symmetries will already be contained (because we only look at a sub-part of the inflation symmetry)
235 m_marginal_symmetries.insert(inf::Symmetry(inf::PartySym(marg_party_sym), inf_sym.get_outcome_sym()));
236 }

239void inf::Marginal::init_marginal_permutations() {
240 ASSERT_EQUAL(m_marginal_permutations.size(), 0)
241
242 if (m_marginal_parties.size() == 0) {
243 m_denom = 1;
244 return;
245 }
246
247 // We expect some marginal permutations to be duplicate in the below loop (because some inflation symmetries
248 // simply leave the marginal invariant), so we use a std::set to remove duplicates on the fly
249 std::set<inf::Marginal::Permutation> marg_perm_set;
250
251 // We first need to find all the transformations (denoted marg_perm in the following)
252 // under the inflation symmetries (\mathbb{G}^p) of the identity marginal permutation (id,id_A)
253 for (inf::Symmetry const &inf_sym : m_inflation->get_inflation_symmetries()) {
254
255 std::vector<Index> transformed_marg_parties(get_n_parties(), 0);
256 inf_sym.get_party_sym().act_entrywise(m_marginal_parties, transformed_marg_parties);
257 inf::Marginal::Permutation const marg_perm(inf_sym.get_outcome_sym(), transformed_marg_parties);
258
259 // By default, marg_perm is a representative of its orbit under marginal symmetries
260 bool is_representative = true;
261
262 // We now need to check if marg_perm is indeed a representative of its orbit
263 for (inf::Symmetry const &marg_sym : m_marginal_symmetries) {
264
265 inf::OutcomeSym const other_outcome_sym = marg_perm.outcome_sym.get_composition_after(marg_sym.get_outcome_sym().get_inverse());
266
267 std::vector<Index> other_marg_parties(get_n_parties(), 0);
268 marg_sym.get_party_sym().act_on_list(transformed_marg_parties, other_marg_parties);
269
270 inf::Marginal::Permutation const other_marg_perm(other_outcome_sym, other_marg_parties);
271
272 if (other_marg_perm < marg_perm) {
273 is_representative = false;
274 break;
275 }
276 }
277
278 if (is_representative)
279 marg_perm_set.insert(marg_perm);
280 }
281
282 m_marginal_permutations = std::vector<inf::Marginal::Permutation>(marg_perm_set.begin(), marg_perm_set.end());
283 m_denom = m_marginal_permutations.size();
284}

306inf::Marginal::Evaluator inf::Marginal::get_evaluator() const {
307 std::vector<Index> party_weights = inf::DualVector::compute_weights(m_marginal_parties.size(),
308 m_inflation->get_network()->get_n_outcomes(),
309 m_store_bounds);
310
311 inf::Marginal::Evaluator::PartyToUpdateRules party_to_update_rules{};
312
313 for (Index inflation_party : util::Range(m_inflation->get_n_parties())) {
314 party_to_update_rules.emplace_back();
315 // The update rules of the inflation_party
316 inf::Marginal::Evaluator::UpdateRules &update_rules = party_to_update_rules.back();
317
318 // For each marginal permutation, we see if changing the outcome of the inflation_party
319 // would have an impact on the corresponding marginal-permutation-extract marginal event
320 for (Index const marg_perm_index : util::Range(m_marginal_permutations.size())) {
321 inf::Marginal::Permutation const &marg_perm = m_marginal_permutations[marg_perm_index];
322
323 for (Index const marginal_party : util::Range(get_n_parties())) {
324 if (marg_perm.parties[marginal_party] == inflation_party) {
325 inf::OutcomeSym::Bare outcome_sym = marg_perm.outcome_sym.get_inverse_bare_sym();
326 // Need to extend the outcome symmetry to account for the unknown outcome,
327 // which does not undergo any transformation when we swap outcomes
328 if (m_store_bounds == inf::DualVector::StoreBounds::yes)
329 outcome_sym.push_back(static_cast<inf::Outcome>(outcome_sym.size()));
330
331 // Changing the outcome of the inflation event at position inflation_party
332 // has an impact on the marg_perm-extract marginal event at position marginal_party
333 update_rules.emplace_back(marg_perm_index, outcome_sym, party_weights[marginal_party]);
334 break;
335 }
336 }
337 }
338 }
339
340 return inf::Marginal::Evaluator(m_inflation->get_n_parties(),
341 get_n_parties(),
342 inf::DualVector::get_outcomes_per_party(m_inflation->get_network()->get_n_outcomes(), m_store_bounds),
343 m_marginal_permutations.size(),
344 party_to_update_rules);
345}
