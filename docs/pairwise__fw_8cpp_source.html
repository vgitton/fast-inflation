24void inf::PairwiseFW::Data::check_health() const {
25 util::logger << "Checking health... ";
26
27 HARD_ASSERT_EQUAL(m_events.size(), m_vertex_count)
28 HARD_ASSERT_EQUAL(m_weights.size(), m_vertex_count)
29 HARD_ASSERT_EQUAL(m_vertices.size(), m_vertex_count)
30 HARD_ASSERT_EQUAL(m_x_dot_vertex.size(), m_vertex_count)
31
32 double const epsilon = 1.0e-10;
33
34 // Check the normalization of the weights as well as the dimnensions of the vertices
35 {
36 double weight_sum = 0.0;
37 for (Index const vertex_i : util::Range(m_vertex_count)) {
38 HARD_ASSERT_EQUAL(m_vertices[vertex_i].size(), m_dimension)
39
40 weight_sum += m_weights[vertex_i];
41 }
42 HARD_ASSERT_LT(std::abs(weight_sum - 1.0), epsilon)
43 }
44
45 // Check that we indeed have x = \sum_i w[i] * v[i]
46 {
47 double x_vs_convex_error = 0.0;
48 for (Index const dim_i : util::Range(m_dimension)) {
49 double convex_component = 0.0;
50 for (Index vertex_i : util::Range(m_vertex_count)) {
51 convex_component += m_weights[vertex_i] * m_vertices[vertex_i][dim_i];
52 }
53 x_vs_convex_error += std::abs(m_x[dim_i] - convex_component);
54 }
55 HARD_ASSERT_LT(x_vs_convex_error, epsilon)
56 }
57
58 // The norm of x
59 HARD_ASSERT_LT(std::abs(m_x_dot_x - util::inner_product(m_x, m_x)), epsilon)
60
61 // Check the dot products
62 for (Index const vertex_i : util::Range(m_vertex_count)) {
63 HARD_ASSERT_LT(std::abs(util::inner_product(m_x, m_vertices[vertex_i]) - m_x_dot_vertex[vertex_i]), epsilon)
64
65 for (Index const vertex_j : util::Range(m_vertex_count)) {
66 HARD_ASSERT_LT(std::abs(util::inner_product(m_vertices[vertex_i], m_vertices[vertex_j]) - get_vertex_dot_vertex(vertex_i, vertex_j)), epsilon)
67 }
68 }
69
70 util::logger << "ok." << util::cr;
71}

115void inf::PairwiseFW::Data::take_pairwise_step(Index i_min, Index i_max) {
116 // 1 - find gamma
117
118 double gamma = 0.0;
119 {
120 double const norm_direction_squared =
121 get_vertex_dot_vertex(i_max, i_max) - 2.0 * get_vertex_dot_vertex(i_max, i_min) + get_vertex_dot_vertex(i_min, i_min);
122
123 if (norm_direction_squared > 1e-20) {
124 // gamma = util::inner_product(m_x, direction) / norm_direction_squared;
125 gamma = (m_x_dot_vertex[i_max] - m_x_dot_vertex[i_min]) / norm_direction_squared;
126 }
127
128 double const gamma_max = m_weights[i_max];
129
130 if (gamma < 0.0)
131 gamma = 0.0;
132 else if (gamma > gamma_max)
133 gamma = gamma_max;
134 }
135
136 // 2 - update x & norm of x
137
138 m_x_dot_x = 0.0;
139 for (Index const dim_i : util::Range(m_dimension)) {
140 m_x[dim_i] -= gamma * (m_vertices[i_max][dim_i] - m_vertices[i_min][dim_i]);
141 m_x_dot_x += m_x[dim_i] * m_x[dim_i];
142 }
143
144 // 3 - update m_x_dot_vertex
145
146 for (Index vertex_i : util::Range(m_vertex_count)) {
147 // <x', v_i> = <x, v_i> - gamma * <d, v_i>
148 // = <x, v_i> - gamma * ( <v_max, v_i> - <v_min, v_i>)
149 m_x_dot_vertex[vertex_i] -= gamma * (get_vertex_dot_vertex(i_max, vertex_i) - get_vertex_dot_vertex(i_min, vertex_i));
150 }
151
152 // 4 - update convex decomposition
153
154 m_weights[i_min] += gamma;
155 m_weights[i_max] -= gamma;
156 if (m_weights[i_max] < inf::PairwiseFW::Data::cleanup_tolerance) {
157 // Careful, doing this can invalidate i_min
158 remove_vertex(i_max);
159 // Here we could in principle renormalize but not needed
160 }
161}

163void inf::PairwiseFW::Data::clean_up_vertices() {
164 bool removed_at_least_one_vertex = false;
165 Index vertex_i = 0;
166 while (vertex_i < m_vertex_count) {
167 if (m_weights[vertex_i] < inf::PairwiseFW::Data::cleanup_tolerance) {
168 remove_vertex(vertex_i);
169 removed_at_least_one_vertex = true;
170 }
171 // We don't increment i here since the new i-th element
172 // is the old back element, which might also need to be clean up
173 else
174 ++vertex_i;
175 }
176
177 // We now need to renormalize
178 double sum = 0.0;
179 for (double weight : m_weights) {
180 sum += weight;
181 }
182
183 if (sum > 1.0 + inf::PairwiseFW::Data::cleanup_tolerance) {
184 THROW_ERROR("The vertex weights have a sum of " + util::str(sum))
185 }
186
187 for (double &weight : m_weights) {
188 weight /= sum;
189 }
190
191 if (removed_at_least_one_vertex)
192 update_x_from_weights();
193}

197void inf::PairwiseFW::Data::remove_vertex(Index vertex_i) {
198 ASSERT_LT(vertex_i, m_vertex_count)
199
200 --m_vertex_count;
201 m_events[vertex_i] = m_events.back();
202 m_events.pop_back();
203 m_weights[vertex_i] = m_weights.back();
204 m_weights.pop_back();
205 m_vertices[vertex_i] = m_vertices.back();
206 m_vertices.pop_back();
207 m_x_dot_vertex[vertex_i] = m_x_dot_vertex.back();
208 m_x_dot_vertex.pop_back();
209
210 // Here don't reduce size of m_vertex_dot_vertex, just update the valid entries
211 const Index last_vertex_i = m_vertex_count;
212 // Suppose we have four vertices, and we remove vertex_i = 1.
213 // The 4x4 matrix of inner products gets updated as follows:
214 // [ - p0 - c0d ]
215 // [ p0 p* p1 d ]
216 // [ - p2 - c1d ]
217 // [ c0d d c1d c*d ]
218 // Legend:
219 // - = nothing happens,
220 // ci = copy to register i,
221 // pi = paste register i,
222 // d = deleted entry
223 for (Index const vertex_j : util::Range(m_vertex_count)) {
224 if (vertex_j == vertex_i)
225 continue;
226
227 double const inner_product = get_vertex_dot_vertex(last_vertex_i, vertex_j);
228 get_vertex_dot_vertex(vertex_i, vertex_j) = inner_product;
229 get_vertex_dot_vertex(vertex_j, vertex_i) = inner_product;
230 }
231 // Update the diagonal entry separately, it doesn't follow the above pattern
232 get_vertex_dot_vertex(vertex_i, vertex_i) = get_vertex_dot_vertex(last_vertex_i, last_vertex_i);
233}

235void inf::PairwiseFW::Data::update_x_from_weights() {
236 m_x_dot_x = 0.0;
237
238 for (Index dim_i : util::Range(m_dimension)) {
239 m_x[dim_i] = 0.0;
240
241 for (Index vertex_i : util::Range(m_vertex_count)) {
242 m_x[dim_i] += m_weights[vertex_i] * m_vertices[vertex_i][dim_i];

244
245 m_x_dot_x += m_x[dim_i] * m_x[dim_i];
246 }
247
248 for (Index vertex_i : util::Range(m_vertex_count)) {
249 m_x_dot_vertex[vertex_i] = util::inner_product(m_x, m_vertices[vertex_i]);

279inf::FrankWolfe::Solution inf::PairwiseFW::solve() {
280 ASSERT_LT(0, m_data.get_vertex_count())
281
282 // m_data.check_health();
283
284 if (m_data.get_vertex_count() == 1) {
285 return get_current_solution();
286 }
287
288 bool first_step = true;
289
290 while (true) {
291 // m_data.check_health();
292
293 if (m_data.get_x_dot_x() < inf::PairwiseFW::inconclusive_tolerance)
294 break;
295
296 Index i_min, i_max;
297 find_min_and_max_inner_products(i_min, i_max);
298 double const pairwise_gap = m_data.get_x_dot_vertex(i_max) - m_data.get_x_dot_vertex(i_min);
299
300 if (first_step and pairwise_gap < m_phi / inf::PairwiseFW::lazy_tolerance)
301 m_phi *= 0.5;
302
303 if (pairwise_gap < m_phi)
304 break;
305
306 m_data.take_pairwise_step(i_min, i_max);
307
308 if (m_store_iterates) {
309 if (first_step)
310 m_iterates.emplace_back(m_data.get_x(), m_last_fw_vertex, true);
311 else
312 m_iterates.emplace_back(m_data.get_x(), std::vector<double>{}, false);
313 }
314
315 first_step = false;
316 }
317
318 return get_current_solution();
319}

368void inf::PairwiseFW::find_min_and_max_inner_products(Index &i_min, Index &i_max) const {
369 // It's not clear what's best between true and false, both make sense for sure
370 bool use_traditional_pairwise_fw = false;
371 if (use_traditional_pairwise_fw) {
372 double min_inner = std::numeric_limits<double>::max();
373 // Don't use min() here!!!
374 double max_inner = std::numeric_limits<double>::lowest();
375
376 ASSERT_LT(0, m_data.get_vertex_count())
377
378 for (Index vertex_i : util::Range(m_data.get_vertex_count())) {
379 double const current_inner = m_data.get_x_dot_vertex(vertex_i);
380
381 if (current_inner > max_inner) {
382 max_inner = current_inner;
383 i_max = vertex_i;
384 }
385
386 if (current_inner < min_inner) {
387 min_inner = current_inner;
388 i_min = vertex_i;
389 }
390
391 ASSERT_LT(std::numeric_limits<double>::lowest(), max_inner)
392 ASSERT_LT(min_inner, std::numeric_limits<double>::max())
393 }
394 } else {
395 // Renormalize
396 double best_score = std::numeric_limits<double>::lowest();
397
398 ASSERT_LT(0, m_data.get_vertex_count())
399
400 Index vertex_i, vertex_j;
401
402 for (vertex_i = 0; vertex_i < m_data.get_vertex_count(); ++vertex_i) {
403 for (vertex_j = 0; vertex_j < m_data.get_vertex_count(); ++vertex_j) {
404 // Note: pairs (vertex_i, vertex_j) are ordered to take the sign into account
405 if (vertex_j == vertex_i)
406 continue;
407
408 double const norm = std::sqrt(
409 m_data.get_vertex_dot_vertex(vertex_i, vertex_i) - 2.0 * m_data.get_vertex_dot_vertex(vertex_i, vertex_j) + m_data.get_vertex_dot_vertex(vertex_j, vertex_j));
410
411 if (norm > 0.0) {
412 double const score = (m_data.get_x_dot_vertex(vertex_i) - m_data.get_x_dot_vertex(vertex_j)) / norm;
413
414 if (score > best_score) {
415 best_score = score;
416 i_min = vertex_j;
417 i_max = vertex_i;
418 }
419 }
420 }
421 }
422 }
423
424 ASSERT_LT(i_min, m_data.get_vertex_count())
425 ASSERT_LT(i_max, m_data.get_vertex_count())
