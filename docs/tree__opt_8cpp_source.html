125inf::TreeOpt::ThreadReturn inf::TreeOpt::thread_opt(std::vector<inf::TreeSplitter::Path> const &paths) {
126 // Initialization of the worker. Don't forget to reset it if we reuse them in the future!
127 inf::TreeOpt::ThreadWorker thread_worker(m_marg_evaluators);
128
129 for (inf::TreeSplitter::Path const &path : paths) {
130 Index const end_of_path = path.size() - 1;
131
132 for (Index const depth : util::Range(m_inflation_n_parties)) {
133
134 if (depth <= end_of_path) {
135 inf::EventTree::NodePos const node_pos(depth, path[depth]);
136
137 if (depth < end_of_path) {
138 // Set the first outcomes according to the path
139 inf::EventTree::Node const &the_node = m_event_tree->get_node(node_pos);
140 thread_worker.marg_evaluators.set_outcome(depth, the_node.outcome);
141 } else if (depth == end_of_path) {
142 // Init queue with a single node to be processed below
143 thread_worker.queue = inf::EventTree::NodePos::Queue({node_pos});
144 }
145 } else { // depth > end_of_path
146 // If we don't store the bounds, we don't care about the end of the event being initialized to "?"
147 if (not m_store_bounds)
148 break;
149 else
150 thread_worker.marg_evaluators.set_outcome(depth, m_outcome_unknown);
151 }
152 }
153
154 while (not thread_worker.queue.empty()) {
155 inf::EventTree::NodePos const node_pos = util::pop_back(thread_worker.queue);
156 go_down_from(thread_worker, node_pos);
157 }
158 }
159
160 return inf::TreeOpt::ThreadReturn(inf::Optimizer::PreSolution(thread_worker.current_minimum,
161 thread_worker.current_best_event),
162 thread_worker.n_leaves_effective);
163}

165void inf::TreeOpt::go_down_from(inf::TreeOpt::ThreadWorker &thread_worker,
166 inf::EventTree::NodePos const &node_pos) {
167
168 // In sat mode, as soon as the global minimum is <= 0, we stop minimizing.
169 // In particular, the <= 0 score may be found in a differen thread.
170 if (m_stop_mode == inf::Optimizer::StopMode::sat and m_global_minimum.get() <= 0) {
171 thread_worker.queue.clear();
172 return;
173 }
174
175 if (m_store_bounds) {
176 // Here thread_worker.last_depth_processed may be m_inflation_n_parties-1 (i.e., unitialized from a
177 // previous path), but it doesn't hurt to set the last outcomes to "?"
178 for (Index depth(node_pos.depth + 1); depth <= thread_worker.last_depth_processed; ++depth)
179 thread_worker.marg_evaluators.set_outcome(depth, m_outcome_unknown);
180
181 thread_worker.last_depth_processed = node_pos.depth;
182 }
183
184 inf::EventTree::Node const &the_node = m_event_tree->get_node(node_pos);
185 thread_worker.marg_evaluators.set_outcome(node_pos.depth, the_node.outcome);
186
187 // if we're not at the last depth
188 if (node_pos.depth < m_inflation_n_parties - 1) {
189 bool keep_branch = true;
190
191 if (m_store_bounds) {
192 Num const score_lower_bound = thread_worker.marg_evaluators.evaluate_dual_vector();
193
194 // We keep the branch only if the lower bound suggests that there is a chance to do strictly better than the current minimum
195 keep_branch = (score_lower_bound < m_global_minimum.get());
196 }
197
198 if (keep_branch)
199 m_event_tree->add_children_to_queue(thread_worker.queue, node_pos);
200 else
201 thread_worker.n_leaves_effective += 1;
202
203 } else // if at the last depth
204 {
205 thread_worker.n_leaves_effective += 1;
206 Num const score = thread_worker.marg_evaluators.evaluate_dual_vector();
207
208 if (score < m_global_minimum.get()) {
209 thread_worker.current_minimum = score;
210 thread_worker.current_best_event = thread_worker.marg_evaluators.get_inflation_event();
211
212 m_global_minimum.set_if_smaller(score);
213 }
214 }
