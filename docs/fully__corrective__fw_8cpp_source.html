137double inf::FullyCorrectiveFW::get_min_inner_product(std::vector<double> const &vec) const {
138 double min = 0.0;
139 bool min_uninit = true;
140
141 for (std::vector<double> const &row : m_matrix) {
142 if (min_uninit) {
143 min = util::inner_product(vec, row);
144 min_uninit = false;
145 } else {
146 double const new_val = util::inner_product(vec, row);
147 if (new_val < min)
