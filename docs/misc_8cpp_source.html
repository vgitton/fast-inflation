37Num dummy_eval_dual_vector_on_event(
38 inf::DualVector const &dual_vector,
39 inf::Event const &e) {
40 inf::Inflation const &infl = *dual_vector.get_inflation();
41
42 ASSERT_EQUAL(infl.get_n_parties(), e.size())
43
44 Index n_alpha = infl.get_size()[0];
45 Index n_beta = infl.get_size()[1];
46 Index n_gamma = infl.get_size()[2];
47
48 Num alt_score = 0;
49 for (Index alpha1 : util::Range(n_alpha)) {
50 for (Index alpha2 : util::Range(alpha1 + 1, n_alpha)) {
51 for (Index beta1 : util::Range(n_beta)) {
52 for (Index beta2 : util::Range(n_beta)) {
53 if (beta1 == beta2)
54 continue;
55 for (Index gamma1 : util::Range(n_gamma)) {
56 for (Index gamma2 : util::Range(n_gamma)) {
57 if (gamma1 == gamma2)
58 continue;
59
60 alt_score += dual_vector.get_event_tensor().get_num({e[infl.get_party_index(
61 inf::Inflation::Party(0, {beta1, gamma1}))],
62 e[infl.get_party_index(
63 inf::Inflation::Party(1, {gamma1, alpha1}))],
64 e[infl.get_party_index(
65 inf::Inflation::Party(2, {alpha1, beta1}))],
66 e[infl.get_party_index(
67 inf::Inflation::Party(0, {beta2, gamma2}))],
68 e[infl.get_party_index(
69 inf::Inflation::Party(1, {gamma2, alpha2}))],
70 e[infl.get_party_index(
71 inf::Inflation::Party(2, {alpha2, beta2}))]});
72 }
73 }
74 }
75 }
76 }
77 }
78
79 return alt_score;
80}

82Num dummy_eval_dual_vector_on_distribution(
83 inf::DualVector const &dual_vector,
84 inf::TargetDistr const &d) {
85 Num score = 0;
86 for (inf::Event const &e1 : d.get_event_range()) {
87 for (inf::Event const &e2 : d.get_event_range()) {
88 score += d.get_event_tensor().get_num(e1) * d.get_event_tensor().get_num(e2) * dual_vector.get_event_tensor().get_num({e1[0], e1[1], e1[2], e2[0], e2[1], e2[2]});
89 }
90 }
91 return score;
92}

97void user::test_dual_vector_eval(inf::TargetDistr::ConstPtr const &d) {
98 util::logger << *(d->get_network()) << util::cr << *d << util::cr;
99
100 // pick random inflation order
101 util::RNG<Index> inf_order_rng(2, 3);
102 Index n_alpha = inf_order_rng.get_rand();
103 Index n_beta = inf_order_rng.get_rand();
104 Index n_gamma;
105 // Order 3x3x3 is a bit slow for a test
106 if (n_alpha == 3 and n_beta == 3) {
107 n_gamma = 2;
108 } else {
109 n_gamma = inf_order_rng.get_rand();
110 }
111
112 Num lhs_denom = util::factorial(n_alpha) * util::factorial(n_beta) * util::factorial(n_gamma) / 2;
113 Num rhs_denom = util::pow(d->get_event_tensor().get_denom(), Num(2));
114 Num gcd = std::gcd(lhs_denom, rhs_denom);
115 lhs_denom /= gcd;
116 rhs_denom /= gcd;
117
118 // Inflation
119 inf::Inflation::ConstPtr inflation = std::make_shared<const inf::Inflation>(
120 d, inf::Inflation::Size({n_alpha, n_beta, n_gamma}));
121 util::logger << *inflation << util::cr;
122
123 // LPI constraint
124 inf::ConstraintSet constraints(inflation, {{"A00,B00,C00", "A11,B11,C11", ""}}, inf::DualVector::StoreBounds::yes);
125 constraints.set_target_distribution(*d);
126
127 // setup some dummy coeffs
128 inf::Quovec dual_vector_quovec(constraints.get_quovec_size());
129 util::RNG<Num> dual_vector_coeff_rng(-10, 10);
130 for (Num &coeff : dual_vector_quovec) {
131 coeff = dual_vector_coeff_rng.get_rand();
132 }
133 constraints.set_dual_vector_from_quovec(dual_vector_quovec);
134 constraints.log_dual_vectors();
135
136 // Inflation event
137 LOG_BEGIN_SECTION("Example inflation event")
138 inf::Event e = inflation->get_random_event();
139 inflation->log_event(e);
140 LOG_END_SECTION
141 util::logger << util::cr;
142
143 // Evaluation
144
145 inf::Marginal::EvaluatorSet evaluators = constraints.get_marg_evaluators();
146 for (Index i : util::Range(e.size())) {
147 evaluators.set_outcome(i, e[i]);
148 }
149 Num const event_score = evaluators.evaluate_dual_vector();
150
151 LOG_BEGIN_SECTION("LPI constraint evaluation")
152
153 util::logger
154 << "Using inf::Marginal::EvaluatorSet::evaluate_dual_vector(): " << util::cr
155 << "\t" << event_score << util::cr;
156
157 // Just to extract the inf::DualVector, a bit dirty but otherwise we need to expose too much of the internals of inf::ConstraintSet
158 inf::Constraint first_constraint(inflation, {"A00,B00,C00", "A11,B11,C11", {}}, inf::DualVector::StoreBounds::no);
159 first_constraint.set_target_distribution(*d);
160 first_constraint.set_dual_vector_from_quovec(dual_vector_quovec, 0);
161
162 Num event_score_dummy =
163 rhs_denom * user::dummy_eval_dual_vector_on_event(first_constraint.get_dual_vector(), e) - lhs_denom * user::dummy_eval_dual_vector_on_distribution(first_constraint.get_dual_vector(), *d);
164
165 // util::logger << "rhs_denom * dual_vector_on_event = " << rhs_denom * user::dummy_eval_dual_vector_on_event(first_constraint->get_dual_vector(), e) << util::cr;
166 // util::logger << "lhs_denom * dual_vector_on_d = " << lhs_denom * user::dummy_eval_dual_vector_on_distribution(first_constraint->get_dual_vector(), *d) << util::cr;

174 constraints.get_inflation_event_quovec(e),

228void user::event_sym::run() {
229 inf::TargetDistr::ConstPtr const ejm = user::get_ejm_distribution();
230 inf::Inflation const inflation(ejm, inf::Inflation::Size{2, 2, 2});
231 inf::Event event(inflation.get_n_parties());
232
233 // Alice: \setinfeventalice{0}{0}{1}{2}
234 event[inflation.get_party_index({0, {0, 0}})] = 0;
235 event[inflation.get_party_index({0, {0, 1}})] = 0;
236 event[inflation.get_party_index({0, {1, 0}})] = 1;
237 event[inflation.get_party_index({0, {1, 1}})] = 2;
238 // Bob: \setinfeventbob{1}{1}{2}{3}
239 event[inflation.get_party_index({1, {0, 0}})] = 1;
240 event[inflation.get_party_index({1, {0, 1}})] = 1;
241 event[inflation.get_party_index({1, {1, 0}})] = 2;
242 event[inflation.get_party_index({1, {1, 1}})] = 3;
243 // Charlie: \setinfeventcharlie{0}{1}{2}{3}
244 event[inflation.get_party_index({2, {0, 0}})] = 0;
245 event[inflation.get_party_index({2, {0, 1}})] = 1;
246 event[inflation.get_party_index({2, {1, 0}})] = 2;
247 event[inflation.get_party_index({2, {1, 1}})] = 3;
248
249 inf::Event const input_event = event;
250
251 util::logger << "Input event: " << util::cr;
252 inflation.log_event(event);
253 user::print_event_latex_format(inflation, event);
254
255 // Outcome sym = (01)
256 inf::OutcomeSym const outcome_sym({1, 0, 2, 3});
257 util::logger << "Applying the outcome symmetry " << outcome_sym
258 << " (exchange outcomes " << inflation.get_network()->outcome_to_str(0)
259 << " and " << inflation.get_network()->outcome_to_str(1)
260 << ") yields:" << util::cr;
261 outcome_sym.act_on_event(event, event);
262 inflation.log_event(event);
263 user::print_event_latex_format(inflation, event);
264
265 // Party sym = (01)
266 inf::PartySym const party_sym({1, 0, 2}, false);
267 util::logger << "Applying the party symmetry " << party_sym << " (exchange Alice and Bob) yields:" << util::cr;
268 inf::PartySym const party_sym_action = inflation.network_party_to_inf_party_sym(party_sym);
269 {
270 inf::Event transformed_event(inflation.get_n_parties());
271 party_sym_action.act_on_list(event, transformed_event);
272 event = transformed_event;
273 }
274 inflation.log_event(event);
275 user::print_event_latex_format(inflation, event);
276
277 // Source sym = ( (01), (01), id)
278 std::vector<std::vector<Index>> const source_sym = {{1, 0}, {1, 0}, {0, 1}};
279 util::logger << "Applying the source symmetry " << source_sym << " (exchange alpha0 and alpha1, exchange beta0 and beta1) yields:" << util::cr;
280 inf::PartySym const source_sym_action = inflation.source_sym_to_party_sym(source_sym);
281 {
282 inf::Event transformed_event(inflation.get_n_parties());
283 source_sym_action.act_on_list(event, transformed_event);
284 event = transformed_event;
285 }
286 inflation.log_event(event);
287 user::print_event_latex_format(inflation, event);
288
289 util::logger << "Alternatively, we can directly act on the input event with the inflation symmetry (source,party,outcome):" << util::cr;
290 inf::Symmetry const full_sym(source_sym_action.get_composition_after(party_sym_action), outcome_sym);
291 inf::Event other_event(inflation.get_n_parties());
292 full_sym.act_on_event(input_event, other_event);
293 inflation.log_event(other_event);
294 user::print_event_latex_format(inflation, other_event);
295
296 HARD_ASSERT_EQUAL(event, other_event)
297
298 util::logger << util::cr;
299}

396void user::pairwise_fw::run() {
397 Index const dimension = 2;
398
399 inf::PairwiseFW blender(dimension);
400 blender.set_store_iterates(true);
401
402 std::vector<std::vector<Num>> vertices = {
403 // {-2, 0},
404 // {+2, 0},
405 // {-1, 1},
406 // {+1, 1}
407 {-3, 3},
408 {3, 3},
409 {1, 2},
410 {-2, 1},
411 {0, 1},
412 {3, 1},
413 {-3, -1},
414 {-1, -1},
415 {2, -1},
416 {0, 4}};
417
418 std::vector<Num> initial_vertex = vertices.back();
419
420 blender.memorize_event_and_quovec({}, initial_vertex, 1.0);
421
422 Index iter = 0;
423 while (true) {
424 ++iter;
425 if (iter == 101) {
426 util::logger << "Reached the max number of iterations. Exiting." << util::cr;
427 break;
428 }
429
430 inf::FrankWolfe::Solution sol = blender.solve();
431 // util::logger << sol;
432
433 if (not sol.valid) {
434 util::logger << "We got close to zero! Exiting." << util::cr;
435 break;
436 }
437
438 std::vector<Num> const *fw_vertex = nullptr;
439 for (std::vector<Num> const &vertex : vertices) {
440 std::vector<double> vertex_double(dimension);
441 for (Index i : util::Range(dimension)) {
442 vertex_double[i] = static_cast<double>(1000000 * vertex[i]);
443 }
444
445 if (util::inner_product(vertex_double, sol.vec) < 0) {
446 fw_vertex = &vertex;
447 break;
448 }
449 }
450
451 if (fw_vertex == nullptr) {
452 util::logger << "Found a separating hyperplane! Exiting." << util::cr;
453 break;
454 }
455
456 blender.memorize_event_and_quovec({}, *fw_vertex, 1.0);
457 }
458
459 util::log_mathematica("vertices", vertices);
460 util::log_mathematica("iterates", blender.get_iterates());
461}

765void user::opt_bounds::run() {
766 inf::TargetDistr::ConstPtr d = user::get_noisy_srb(1, 2);
767
768 // Inflation
769 inf::Inflation::Size const inf_size = {2, 2, 2};
770 inf::Inflation::ConstPtr inflation = std::make_shared<const inf::Inflation>(d, inf_size);
771 util::logger << *inflation << util::cr;
772
773 inf::Optimizer::SearchMode const search_mode = inf::Optimizer::SearchMode::tree_search;
774
775 std::vector<inf::DualVector::StoreBounds> store_bound_modes = {
776 inf::DualVector::StoreBounds::yes,
777 inf::DualVector::StoreBounds::no,
778 };
779
780 inf::Quovec dual_vector_quovec;
781
782 std::unique_ptr<Num> previous_score = nullptr;
783
784 for (inf::DualVector::StoreBounds store_bounds : store_bound_modes) {
785 inf::DualVector::log(store_bounds);
786 util::logger << util::cr << util::cr;
787
788 // LPI constraint
789 inf::ConstraintSet::Ptr constraints = std::make_shared<inf::ConstraintSet>(inflation,
790 inf::ConstraintSet::Description{{"A00,B00,C00", "A11,B11,C11", {}}},
791 store_bounds);
792
793 constraints->set_target_distribution(*d);
794
795 inf::Optimizer::Ptr optimizer = inf::Optimizer::get_optimizer(search_mode, constraints, inf::EventTree::IO::none, get_feas_options()->get_n_threads());
796
797 // This mechanism sets the dual_vector_quovec during the first pass, but leaves it intact in the second pass so that we can compare
798 // the optimization of the same dual_vector
799 if (dual_vector_quovec.size() == 0) {
800 dual_vector_quovec = inf::Quovec(constraints->get_quovec_size());
801 util::RNG<Num> dual_vector_coeff_rng(-10, 10);
802 // Setup some dummy coeffs
803 for (Num &coeff : dual_vector_quovec)
804 coeff = dual_vector_coeff_rng.get_rand();
805 }
806
807 constraints->set_dual_vector_from_quovec(dual_vector_quovec);
808
809 inf::Optimizer::Solution sol = optimizer->optimize(inf::Optimizer::StopMode::opt);
810
811 if (previous_score != nullptr) {
812 HARD_ASSERT_EQUAL(*previous_score, sol.get_inflation_event_score())
813 util::logger << "The inner products are equal irrespective of whether or one stores the dual bounds." << util::cr;
814 }
815
816 previous_score = std::make_unique<Num>(sol.get_inflation_event_score());
817
818 util::logger << util::cr;
819 }

822void user::opt_time::run() {
823 inf::TargetDistr::ConstPtr d = user::get_ejm_distribution();
824
825 // Setup
826 inf::Optimizer::SearchMode const search_mode = inf::Optimizer::SearchMode::tree_search;
827 inf::DualVector::StoreBounds const store_bounds = inf::DualVector::StoreBounds::yes;
828 util::logger << "Timing an inf::Optimizer initialized with ";
829 inf::Optimizer::log(search_mode);
830 util::logger << " and ";
831 inf::DualVector::log(store_bounds);
832 util::logger << util::cr;
833
834 // Inflation
835 inf::Inflation::Size inf_size = {2, 2, 3};
836 inf::Inflation::ConstPtr inflation = std::make_shared<const inf::Inflation>(d, inf_size);
837 util::logger << *inflation << util::cr;
838
839 // LPI constraint
840 inf::ConstraintSet::Ptr constraints = std::make_shared<inf::ConstraintSet>(
841 inflation,
842 inf::ConstraintSet::Description{{"A00,B00,C00", "A11,A12,B11,B21,C11"}},
843 store_bounds);
844 constraints->set_target_distribution(*d);
845
846 // Optimizer
847 inf::Optimizer::Ptr optimizer = inf::Optimizer::get_optimizer(search_mode, constraints, get_feas_options()->get_symtree_io(), get_feas_options()->get_n_threads());
848
849 // Now we time!
850 Index n_runs = 20;
851 util::Chrono chrono(util::Chrono::State::paused);
852 for (Index time_i : util::Range(n_runs)) {
853 (void)time_i;
854
855 inf::Quovec dual_vector_quovec(constraints->get_quovec_size());
856 util::RNG<Num> dual_vector_coeff_rng(-10, 10);
857 // Setup some dummy coeffs
858 for (Num &coeff : dual_vector_quovec) {
859 coeff = dual_vector_coeff_rng.get_rand();
860 }
861 constraints->set_dual_vector_from_quovec(dual_vector_quovec);
862
863 chrono.start();
864 inf::Optimizer::Solution sol = optimizer->optimize(inf::Optimizer::StopMode::opt);
865 (void)sol;
866 chrono.pause();
867 }
868
869 double s_per_call = chrono.get_seconds() / static_cast<double>(n_runs);
870 LOG_VARIABLE(s_per_call)
871 double s_per_call_per_leaf = s_per_call / static_cast<double>(inflation->get_symtree(inf::EventTree::IO::none).get_n_leaves());
872 LOG_VARIABLE(s_per_call_per_leaf)
873}
