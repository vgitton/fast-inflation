15inf::TargetDistr::ConstPtr user::get_symmetric_distribution(Index s111,
16 Index s112,
17 Index s123,
18 Index s_denom) {
19 HARD_ASSERT_EQUAL(s111 + s112 + s123, s_denom)
20
21 inf::Network::ConstPtr network = user::get_ejm_network();
22
23 inf::EventTensor d(network->get_n_parties(), network->get_n_outcomes());
24
25 // Smallest common multiple (4,36,24) = (2^2, 2^2*3^2, 2^3*3) = 2^3*3^2 = 72
26 Num const d_denom = s_denom * 72;
27 // Have:
28 // 1/4 = 18/72
29 // 1/36 = 2/72
30 // 1/24 = 3/72
31
32 d.set_denom(d_denom);
33
34 for (inf::Event const &e : network->get_event_range()) {
35 if (e[0] == e[1] and e[1] == e[2]) {
36 d.get_num(e) = s111 * 18;
37 } else if (e[0] != e[1] and e[1] != e[2] and e[2] != e[0]) {
38 d.get_num(e) = s123 * 3;
39 } else {
40 d.get_num(e) = s112 * 2;
41 }
42 }
43 return std::make_shared<inf::TargetDistr>(
44 "(s111,s112,s123) = (" + util::str(s111) + "," + util::str(s112) + "," + util::str(s123) + ") / " + util::str(s_denom),
45 "EJM",
46 network,
47 d);
48}

50inf::TargetDistr::ConstPtr user::get_noisy_pureejm(Num vis, Num vis_denom) {
51 HARD_ASSERT_LT(vis, vis_denom + 1)
52
53 inf::Network::ConstPtr network = user::get_ejm_network();
54
55 inf::EventTensor d(network->get_n_parties(), network->get_n_outcomes());
56
57 // Mixed:
58 // p(a,b,c) = 1/64 = 3/192
59 // Pure EJM:
60 // p(000) = 1/8 = 24/192, p(012) = 1/48 = 4/192
61 // Common denom: 192 * vis_denom
62 Num const d_denom = vis_denom * 192;
63
64 d.set_denom(d_denom);
65
66 for (inf::Event const &e : network->get_event_range()) {
67 // 111
68 if (e[0] == e[1] and e[1] == e[2]) {
69 d.get_num(e) = (vis_denom - vis) * 3 + vis * 24;
70 }
71 // 123
72 else if (e[0] != e[1] and e[1] != e[2] and e[2] != e[0]) {
73 d.get_num(e) = (vis_denom - vis) * 3 + vis * 4;
74 }
75 // 112
76 else {
77 d.get_num(e) = (vis_denom - vis) * 3 + vis * 0;
78 }
79 }
80
81 std::string const name = "Noisy purified EJM with vis " + inf::VisProblem::visibility_to_str(vis, vis_denom);
82
83 return std::make_shared<inf::TargetDistr>(name, "EJM", network, d);
84}

88
89 inf::EventTensor d(network->get_n_parties(), network->get_n_outcomes());

100 }

111// We use this function to print out the first few and the last of the symmetrized
112// inflation events in the paper. This is designed to work only for the 222 inflation.
113void print_some_events(inf::Inflation const &inflation) {
114 for (Index const size : inflation.get_size())
115 HARD_ASSERT_EQUAL(size, 2)
116
117 util::logger << " %%%%%%%%%%%%%%%%%%%%% LaTeX PRINTING %%%%%%%%%%%%%%%%%%%%% " << util::cr;
118
119 std::vector<std::string> const names = {"alice", "bob", "charlie"};
120 util::logger << "Ordering:" << util::cr;
121 util::logger << "(";
122 for (Index const party : util::Range(inflation.get_n_parties())) {
123 if (party > 0)
124 util::logger << ",";
125 util::logger << "\\infevent(\\"
126 << names[inflation.get_parties()[party].first]
127 << "{" << inflation.get_parties()[party].second[0]
128 << inflation.get_parties()[party].second[1] << "})";
129 }
130 util::logger << ")" << util::cr << util::cr;
131
132 inf::EventTree const &tree = inflation.get_symtree(inf::EventTree::IO::none);
133
134 inf::EventTree::NodePos::Queue queue = tree.get_root_children_queue();
135 Index leaf_count = 0;
136 Index const n_repr = 8;
137 Index repr_index = n_repr - 1;
138 std::vector<inf::Event> event_reprs(n_repr, inf::Event(inflation.get_n_parties()));
139 inf::Event working_event(inflation.get_n_parties());
140
141 while (not queue.empty()) {
142
143 inf::EventTree::NodePos const node_pos = util::pop_back(queue);
144 working_event[node_pos.depth] = tree.get_node(node_pos).outcome;
145
146 if (node_pos.depth < inflation.get_n_parties() - 1)
147 tree.add_children_to_queue(queue, node_pos);
148 else {
149 if (leaf_count == 0 or leaf_count >= tree.get_n_leaves() - (n_repr - 1)) {
150 event_reprs[repr_index] = working_event;
151 --repr_index;
152 }
153 leaf_count += 1;
154 }
155 }
156
157 // Print the events
158 for (inf::Event const &event : event_reprs) {
159 // Print the event
160 inflation.log_event(event);
161 // Print the event in latex
162 user::print_event_latex_format(inflation, event);
163 }
164
165 util::logger << "Orbit sample:" << util::cr << util::cr;
166
167 // Print the orbits
168 for (Index const &i : util::Range<Index>(2)) {
169 inf::Event const &event_repr = event_reprs[i];
170 inflation.log_event(event_repr);
171 user::print_event_latex_format(inflation, event_repr);
172
173 std::set<inf::Event> orbit;
174
175 for (inf::Symmetry const &symmetry : inflation.get_inflation_symmetries()) {
176 symmetry.act_on_event(event_repr, working_event);
177 orbit.insert(working_event);
178 }
179
180 Index j = 0;
181 std::set<inf::Event>::const_iterator orbit_it = orbit.begin();
182 while (j < 4 and j < orbit.size()) {
183 inflation.log_event(*orbit_it);
184 user::print_event_latex_format(inflation, *orbit_it);
185 ++orbit_it;
186 ++j;
187 }
188
189 util::logger << "..." << util::cr;
190
191 inflation.log_event(*(--orbit.end()));
192 user::print_event_latex_format(inflation, *(--orbit.end()));
193 }
194
195 util::logger << " %%%%%%%%%%%%%%%%%%%%% END LaTeX PRINTING %%%%%%%%%%%%%%%%%%%%% " << util::cr;
196}

574void user::scan_symmetric_distributions(inf::FeasOptions::Ptr const &feas_options,
575 Index const s_denom,
576 Index const run_index,
577 Index const n_runs) {
578 HARD_ASSERT_LT(run_index, n_runs)
579
580 Index const n_distributions = (s_denom + 1) * (s_denom + 2) / 2;
581
582 Index const start_distribution_index = run_index * n_distributions / n_runs;
583
584 Index const end_distribution_index = static_cast<Index>((run_index + 1) * n_distributions / n_runs - 1);
585
586 LOG_BEGIN_SECTION_FUNC
587
588 inf::TargetDistr::ConstPtr d = user::get_symmetric_distribution(s_denom, 0, 0, s_denom);
589
590 util::logger << "s_denom: " << s_denom << ", number of distributions: " << n_distributions << util::cr;
591 util::logger << "run index: " << run_index << ", n_runs: " << n_runs
592 << ", scanning distributions #" << start_distribution_index
593 << " through #" << end_distribution_index << util::cr;
594
595 util::logger << *feas_options;
596
597 inf::FeasProblem::RetainEvents const retain_events = inf::FeasProblem::RetainEvents::yes;
598 inf::FeasProblem::log(retain_events);
599 util::logger << util::cr;
600
601 inf::FeasProblem feas_problem(d, feas_options);
602
603 std::string csv_results = "\n\n";
604 csv_results += "(feas = 0 means nonlocal, feas = 1 means inconclusive)\n";
605 csv_results += "----------------------------------\n";
606 csv_results += "lambda111,lambda112,lambda123,feas\n";
607
608 Index current_distribution_index = 0;
609
610 LOG_BEGIN_SECTION("Scanning")
611 for (Index const s111 : util::Range(s_denom + 1)) {
612 for (Index const s112 : util::Range(static_cast<Index>(s_denom - s111 + 1))) {
613 bool const out_of_range = (current_distribution_index < start_distribution_index or current_distribution_index > end_distribution_index);
614
615 ++current_distribution_index;
616
617 if (out_of_range)
618 continue;
619
620 Index const s123 = static_cast<Index>(s_denom - s111 - s112);
621
622 d = user::get_symmetric_distribution(s111, s112, s123, s_denom);
623
624 feas_problem.update_target_distribution(d, retain_events);
625
626 inf::FeasProblem::Status const feas_status = feas_problem.get_feasibility();
627
628 csv_results += util::str(float(s111) / float(s_denom)) + "," +
629 util::str(float(s112) / float(s_denom)) + "," +
630 util::str(float(s123) / float(s_denom)) + "," +
631 (feas_status == inf::FeasProblem::Status::nonlocal ? "0" : "1") + "\n";
632 }
633 }
634 HARD_ASSERT_EQUAL(current_distribution_index, n_distributions)
