41inf::Symmetry::Group inf::Constraint::get_constraint_group(inf::Inflation const &inflation,
42 std::vector<Index> const &lhs_marg,
43 std::vector<Index> const &rhs_marg) {
44 ASSERT_LT(0, lhs_marg.size())
45
46 // Will be returned
47 inf::Symmetry::Group constraint_group;
48
49 std::set<Index> const B_parties(rhs_marg.begin(), rhs_marg.end());
50
51 std::set<Index> A_parties;
52 for (Index const party_index : lhs_marg) {
53 if (not B_parties.contains(party_index))
54 A_parties.insert(party_index);
55 }
56
57 for (inf::Symmetry const &inf_sym : inflation.get_inflation_symmetries()) {
58 // inf_sym needs to leave the union A_0, ..., A_{k-1} invariant
59 std::set<Index> transformed_A_parties;
60 for (Index const party_index : A_parties)
61 transformed_A_parties.insert(inf_sym.get_party_sym().get_bare_sym()[party_index]);
62
63 if (transformed_A_parties != A_parties)
64 continue;
65
66 // inf_sym also needs to leave B invariant
67 std::set<Index> transformed_B_parties;
68 for (Index const party_index : B_parties)
69 transformed_B_parties.insert(inf_sym.get_party_sym().get_bare_sym()[party_index]);
70
71 if (transformed_B_parties != B_parties)
72 continue;
73
74 // inf_sym has the right property, so it goes in the constraint group
75 constraint_group.insert(inf_sym);
76 }
77
78 return constraint_group;
79}

247void inf::Constraint::set_target_distribution(inf::TargetDistr const &d) {
248 // First, check that the symmetry group of the distribution hasn't changed
249 // This is slightly sub-optimal once we start having several Constraints,
250 // but this is anyway only ran on changing the target distribution
251 // (plus it's very quick)
252 if (not m_inflation->has_symmetries_compatible_with(d)) {
253 throw inf::Symmetry::SymmetriesHaveChanged();
254 }
255
256 m_target_distribution_fixed = true;
257
258 std::vector<inf::EventTensor const *> target_marginals;
259 for (inf::TargetDistr::MarginalName const &marginal_name : m_target_marginal_names) {
260 target_marginals.push_back(&d.get_marginal(marginal_name));
261 }
262
263 m_rhs_target_tensor->set_to_tensor_product(target_marginals);
264
265 update_rhs_reduced_dual_vector();
266}
