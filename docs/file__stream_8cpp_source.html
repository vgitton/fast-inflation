130void util::OutputFileStream::io_str(std::string const &s) {
131 switch (get_format()) {
132 case util::FileStream::Format::binary:
133 io(s.size());
134 io_bytes(s.data(), s.size());
135 break;
136 case util::FileStream::Format::text:
137 // We can't output strings that contains newline, because we use this as a separator

142 break;
