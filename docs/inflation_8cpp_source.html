247inf::PartySym inf::Inflation::source_sym_to_party_sym(std::vector<std::vector<Index>> const &perm_sources) const {
248 // Initialize to values that are known to be too high to check at the end that we initialized each entry
249 inf::PartySym::Bare party_sym(this->get_n_parties(), this->get_n_parties() + 1);
250 Index const network_n_parties = m_target_distr->get_network()->get_n_parties();
251
252 for (Index const network_party_index : util::Range(network_n_parties)) {
253 std::vector<Index> const &perm_left = perm_sources[(network_party_index + 1) % network_n_parties];
254 std::vector<Index> const &perm_right = perm_sources[(network_party_index + 2) % network_n_parties];
255
256 for (Index s_left : util::Range(perm_left.size())) {
257 for (Index s_right : util::Range(perm_right.size())) {
258 // Reasoning: for {0,1,2} -> {1,2,0}, we want to store
259 // {1,2,0}, i.e., {sigma(0), sigma(1), sigma(2)}
260 party_sym[this->get_party_index(inf::Inflation::Party(
261 network_party_index, {s_left, s_right}))] =
262 this->get_party_index(inf::Inflation::Party(
263 network_party_index, {perm_left[s_left], perm_right[s_right]}));
264 }
265 }
266 }
267
268 DEBUG_RUN(
269 for (Index party_sym_element
270 : party_sym){
271 ASSERT_LT(party_sym_element, this->get_n_parties() + 1)})
272
273 return inf::PartySym(party_sym);
274}

276inf::PartySym inf::Inflation::network_party_to_inf_party_sym(
277 inf::PartySym const &network_party_sym) const {
278 inf::PartySym::Bare inf_sym(this->get_n_parties(), this->get_n_parties() + 1);
279 inf::PartySym::Bare net_sym(network_party_sym.get_bare_sym());
280
281 for (Index net_index : util::Range(net_sym.size())) {
282 for (Index s_left : util::Range(m_size[(net_index + 1) % 3])) {
283 for (Index s_right : util::Range(m_size[(net_index + 2) % 3])) {
284 inf::Inflation::Sources s_image({s_left, s_right});
285 if (not network_party_sym.is_even()) {
286 // Transpose if odd permutation
287 // Tr[ABC] = Tr[B^T A^T C^T]
288 s_image = {s_right, s_left};
289 }
290
291 inf_sym[this->get_party_index(inf::Inflation::Party(
292 net_index, {s_left, s_right}))] = this->get_party_index(inf::Inflation::Party(net_sym[net_index], s_image));
293 }
294 }
295 }
296
297 DEBUG_RUN(
298 for (Index inf_sym_element
299 : inf_sym){
300 ASSERT_LT(inf_sym_element, this->get_n_parties() + 1)})
301
302 return inf::PartySym(inf_sym);

331bool inf::Inflation::are_d_separated(std::vector<Index> const &marg_1,
332 std::vector<Index> const &marg_2) const {

334 std::set<inf::Inflation::Source> const parents_2 = get_parents(marg_2);

358bool inf::Inflation::is_injectable_set(std::vector<Index> const &marg) const {
359 // The empty set is not considered injectable
360 if (marg.size() == 0)
361 return false;
362
363 std::set<Index> network_parties;
364 for (Index const type : util::Range(get_network()->get_n_parties()))
365 network_parties.insert(get_party_index(inf::Inflation::Party(type, {0, 0})));
366
367 std::vector<Index> transformed_marg(marg.size());
368 for (inf::PartySym const &source_sym : get_source_induced_syms()) {
369 source_sym.act_entrywise(marg, transformed_marg);
370
371 bool transformed_marg_in_network_parties = true;
372 for (Index const transformed_party : transformed_marg) {
373 if (not network_parties.contains(transformed_party)) {
374 transformed_marg_in_network_parties = false;
375 break;
376 }
377 }
378
379 if (transformed_marg_in_network_parties)
380 return true;
381 }
382
383 return false;
384}

388void inf::Inflation::init_parties() {
389 // First, the inflation size steps
390
391 std::vector<inf::Inflation::Size> inf_size_steps = {{1, 1, 1}};
392 inf::Inflation::Size const &final_inf_size = m_size;
393
394 while (inf_size_steps.back() != final_inf_size) {
395 // Copy here
396 inf::Inflation::Size current_inf_size = inf_size_steps.back();
397
398 for (Index source_i : util::Range(final_inf_size.size())) {
399 if (current_inf_size[source_i] < final_inf_size[source_i]) {
400 current_inf_size[source_i] += 1;
401 inf_size_steps.push_back(current_inf_size);
402 }
403 }
404 }
405
406 Index party_index = 0;
407 for (inf::Inflation::Size const &inf_size : inf_size_steps) {
408 for (inf::Inflation::Sources const &sources : util::ProductRange(inf_size)) {
409 for (Index network_party_i : this->get_network()->get_party_range()) {
410 Index left = sources[(network_party_i + 1) % 3];
411 Index right = sources[(network_party_i + 2) % 3];
412
413 inf::Inflation::Party party(network_party_i, {left, right});
414
415 if (not m_party_map.contains(party)) {
416 m_party_map[party] = party_index;
417 m_parties.push_back(party);
418
419 std::string name = m_target_distr->get_network()->get_party_names()[network_party_i] + util::str(left) + util::str(right);
420 m_name_map[name] = party_index;
421 m_party_names.push_back(name);
422
423 ++party_index;
424 }
425 }
426 }
427 }
428
429 m_n_parties = party_index;
430
431 ASSERT_EQUAL(party_index, get_n_parties())
432 ASSERT_EQUAL(m_party_map.size(), get_n_parties())
433 ASSERT_EQUAL(m_name_map.size(), get_n_parties())
434 ASSERT_EQUAL(m_party_names.size(), get_n_parties())
435}

453void inf::Inflation::init_inflation_symmetries(inf::Symmetry::Group const &distribution_sym_group) {
454 // This is only valid for the triangle
455 HARD_ASSERT_EQUAL(m_size.size(), 3)
456 // Should always have at least the identity
457 HARD_ASSERT_LT(0, distribution_sym_group.size())
458
459 // std::vector<Index> applicable_distribution_syms;
460 inf::Symmetry::Group applicable_distribution_syms;
461
462 switch (m_use_distr_symmetries) {
463 case inf::Inflation::UseDistrSymmetries::yes:
464 applicable_distribution_syms = inf::Inflation::get_applicable_symmetries(distribution_sym_group,
465 m_size);
466 break;
467
468 case inf::Inflation::UseDistrSymmetries::no:
469 applicable_distribution_syms = m_target_distr->get_network()->get_trivial_symmetry_group();
470 break;
471
472 default:
473 THROW_ERROR("switch")
474 }
475
476 // An extra check
477 DEBUG_RUN(
478 Index const expected_size = applicable_distribution_syms.size() * util::factorial(m_size[0]) * util::factorial(m_size[1]) * util::factorial(m_size[2]);)
479
480 // All the party symmetries induced by inflation source exchanges
481 std::vector<inf::PartySym> const &source_syms = this->get_source_induced_syms();
482
483 for (inf::Symmetry const &network_sym : applicable_distribution_syms) {
484 inf::PartySym const inf_party_sym = this->network_party_to_inf_party_sym(network_sym.get_party_sym());
485 inf::OutcomeSym const &outcome_sym = network_sym.get_outcome_sym();
486
487 for (inf::PartySym const &source_induced_party_sym : source_syms) {
488 // It is here equivalent to take the left or right product,
489 // i.e., individual elements are different but the whole group
490 // that one obtains is the same
491 inf::PartySym combined_party_sym = source_induced_party_sym.get_composition_after(inf_party_sym);
492
493 DEBUG_RUN(auto status =)
494 m_inflation_symmetries.insert(inf::Symmetry(combined_party_sym, outcome_sym));
495 ASSERT_TRUE(status.second)
496 }
497 }
498
499 ASSERT_EQUAL(m_inflation_symmetries.size(), expected_size)
500}

502inf::Symmetry::Group inf::Inflation::get_applicable_symmetries(inf::Symmetry::Group const &distribution_sym_group,
503 inf::Inflation::Size const &size) {
504 inf::Symmetry::Group applicable_distribution_syms;
505
506 // This works for the triangle networks, but not for general networks
507 // The point is that exchanging parties is equivalent to exchanging sources
508 // in a very similar way
509
510 inf::Inflation::Size image_size(3, 0);
511
512 for (inf::Symmetry const &network_sym : distribution_sym_group) {
513 network_sym.get_party_sym().act_on_list(size, image_size);
514
515 if (size == image_size) {
516 DEBUG_RUN(auto status =)
517 applicable_distribution_syms.insert(network_sym);
518
519 ASSERT_TRUE(status.second)
520 }
521 }
522
523 return applicable_distribution_syms;
