64std::vector<Index> inf::TreeFiller::find_children(Index depth) {
65 ASSERT_LT(depth, m_inflation_n_parties)
66 ASSERT_EQUAL(m_working_event[depth], m_unknown_outcome)
67
68 // Will be returned
69 std::vector<Index> child_nodes;
70
71 std::vector<Index> const &current_syms = m_current_syms[depth];
72 bool const not_at_last_depth = (depth < m_inflation_n_parties - 1);
73 // We use forward declarations, this is maybe a bit faster
74 Index depth_bis;
75 // This will only be relevant if we're not at last depth
76 std::vector<Index> *next_syms = nullptr;
77 if (not_at_last_depth)
78 next_syms = &m_current_syms[depth + 1];
79 inf::OutcomeSym::Bare const *outcome_sym;
80 inf::PartySym::Bare const *inverse_party_sym;
81 inf::Outcome base_outcome, transformed_outcome;
82 inf::Outcome &outcome_to_fill = m_working_event[depth];
83
84 // We fill outcome_to_fill = m_working_event[depth] with all possible values,
85 // and for each value, we check whether or not m_working_event is symmetrized
86 for (outcome_to_fill = 0; outcome_to_fill < m_n_outcomes; ++outcome_to_fill) {
87 // By default, m_working_event is symmetrized, but if we find a symmetry that
88 // proves it is not symmetrized, is_symmetrized will be set to false
89 bool is_symmetrized = true;
90
91 if (not_at_last_depth)
92 next_syms->clear();
93
94 // This loop is in charge of verifying whether or not m_working_event is symmetrized,
95 // which is stored in is_symmetrized, and it furthermore adequately initializes the
96 // next_syms for the next recursive call
97 m_sym_chrono.start();
98 for (Index const sym_index : current_syms) {
99 outcome_sym = &m_inflation_outcome_symmetries[sym_index];
100 inverse_party_sym = &m_inflation_inverse_party_symmetries[sym_index];
101
102 // We want to compare event_bis := party_sym(outcome_sym(m_working_event))
103 // with m_working_event with respect to the lexicographic ordering.
104 // If event_bis < m_working_event, then m_working_event is not symmetrized
105 // no matter how we will fill it, and we dismiss m_working_event.
106 // If event_bis > m_working_event, then it can be that the current symmetry indexed
107 // by sym_index will not help to symmetrize the event no matter how we fill it
108 // and we can hence dismiss sym_index from the next_syms.
109 for (depth_bis = 0; depth_bis < m_inflation_n_parties; ++depth_bis) {
110 // This is (party_sym(m_working_event))[depth_bis] = m_working_event[inverse_party_sym[depth_bis]]
111 transformed_outcome = m_working_event[(*inverse_party_sym)[depth_bis]];
112
113 // This happens when e.g. m_working_event = (2,?,?,...) and party_sym(m_working_event) = (?,2,?,...)
114 // In this case, event_bis > m_working_event, but we cannot say whether
115 // or not the sym_index will help to symmetrize m_working_event when it
116 // will be filled more, so we add sym_index to the next_syms.
117 // Furthermore, we do not know whether or not m_working_event is symmetrized
118 // (if all outcome symmetries are there, (0,?,?,...) will be the symmetrized one),
119 // so we do not touch is_symmetrized before breaking
120 //
121 // The other case where this can happen is if e.g., m_working_event = (0,0,?)
122 // and the symmetry index by sym_index is just the outcome swap of 2 and 3.
123 // The above reasoning applies equally well in that case.
124 if (transformed_outcome == m_unknown_outcome) {
125 if (not_at_last_depth)
126 next_syms->push_back(sym_index);
127
128 break;
129 }
130
131 base_outcome = m_working_event[depth_bis];
132 ASSERT_LT(base_outcome, m_unknown_outcome)
133
134 // We now have transformed_outcome = party_sym(outcome_sym(m_working_event))[depth_bis]
135 transformed_outcome = (*outcome_sym)[transformed_outcome];
136
137 // If the transformed event is "lower" than the working event, exit the loop concluding
138 // that m_working_event is not symmetrized. In this case, no need to update the next_syms,
139 // because there is anyway no child of this outcome (it is dismissed because not symmetrized)
140 if (transformed_outcome < base_outcome) {
141 is_symmetrized = false;
142 break;
143 }
144
145 // In this case, the symmetry transformation takes the event up in its orbit
146 // This will remain true when filling it further.
147 // This symmetry is thus "irrelevant" for future depths: it has no chance
148 // of symmetrizing the event. We don't add it to next_syms.
149 if (transformed_outcome > base_outcome)
150 break;
151 }
152
153 // If this event is not symmetrized, we can stop. Note that in this case, the value of next_syms
154 // is irrelevant because this node will have no children.
155 if (not is_symmetrized)
156 break;
157 }
158 m_sym_chrono.pause();
159
160 // We dismiss m_working_event with the current outcome_to_fill based on symmetry
161 if (not is_symmetrized)
162 continue;
163
164 std::vector<Index> children{};
165 if (not_at_last_depth) {
166 children = this->find_children(depth + 1);
167
168 // Dismiss empty childset, unless we're at last depth
169 // This can happen if e.g. (0,1,?) looks symmetrized,
170 // but somehow all the filling (0,1,x) are not symmetrized.
171 if (children.size() == 0)
172 continue;
173 }
174
175 // We can now construct the node!
176 m_node_chrono.start();
177 child_nodes.push_back(m_tree.insert_node(depth, outcome_to_fill, children));
178 m_node_chrono.pause();
179 }
180
181 // We need to reset m_working_event[depth] to "?" so that the outer level
182 // of the recursion still has an event where depth+1,depth+2 etc have all "?" outcomes
183 outcome_to_fill = m_unknown_outcome;
184
185 return child_nodes;
186}
