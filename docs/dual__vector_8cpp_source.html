69void inf::DualVector::set_from_quovec(inf::Quovec const &quovec, inf::QuovecIndex start_pos) {
70 // If we have a scalar dual_vector, expect a single coefficient
71 if (get_n_parties() == 0) {
72 // We could in principle allow to read the coefficient in the list,
73 // but that's not the use case we have.
74 // Feel free to generalize the below if this changes.
75 ASSERT_EQUAL(start_pos, 0)
76 ASSERT_EQUAL(quovec.size(), 1)
77
78 m_event_tensor.get_num(0) = quovec[0];
79
80 return;
81 }
82
83 ASSERT_LT(m_n_orbits_no_unknown, quovec.size() - start_pos + 1);
84
85 // Directly set the orbit coeffs with no unknowns
86 for (inf::QuovecIndex const quovec_index : util::Range(m_n_orbits_no_unknown)) {
87 set_orbit_coeff(quovec_index, quovec[start_pos + quovec_index]);
88 }
89
90 // The rest is only executed when we store the bounds
91 if (m_store_bounds == inf::DualVector::StoreBounds::yes) {
92 inf::Quovec quovec_with_unknown(get_n_orbits_with_unknown());
93 for (Index const i : util::Range(m_n_orbits_no_unknown)) {
94 quovec_with_unknown[i] = quovec[start_pos + i];
95 }
96
97 for (BoundRule const &bound_rule : m_bound_rules) {
98
99 // Easter egg, this value is unused
100 Num bound = 42;
101
102 for (Index i : util::Range(bound_rule.second.size())) {
103 Num const potential_bound = quovec_with_unknown[bound_rule.second[i]];
104
105 if (i == 0)
106 bound = potential_bound;
107 else
108 bound = min_or_max(bound, potential_bound);
109 }
110
111 quovec_with_unknown[bound_rule.first] = bound;
112
113 set_orbit_coeff(bound_rule.first, bound);
114 }
115 }

118void inf::DualVector::hard_assert_within_bound(Num const bound) const {
119 for (inf::Event const &orbit_repr : get_orbit_repr_no_unknown()) {
120 Num const component = get_event_tensor().get_num(orbit_repr);
121 if (component >= bound or component <= -bound) {
122 THROW_ERROR(

189void inf::DualVector::init_orbits(inf::Symmetry::Group const &marginal_symmetries) {
190 // Set the symmetries depending on the bounds: if we store the bounds,
191 // we need to account for the trivial action of the outcome symmetries
192 // on the unknown outcome
193 if (m_store_bounds == inf::DualVector::StoreBounds::yes) {
194 inf::Symmetry::Group const &group_no_unknown = marginal_symmetries;
195 inf::Symmetry::Group group_with_unknown;
196
197 const inf::Outcome unknown_outcome = m_inflation->get_network()->get_outcome_unknown();
198
199 for (inf::Symmetry const &sym_no_unknown : group_no_unknown) {
200 inf::OutcomeSym::Bare bare_outcome_sym = sym_no_unknown.get_outcome_sym().get_bare_sym();
201 // The outcome sym sends unknown_outcome to unknown_outcome
202 bare_outcome_sym.push_back(unknown_outcome);
203
204 group_with_unknown.insert(inf::Symmetry(sym_no_unknown.get_party_sym(),
205 inf::OutcomeSym(bare_outcome_sym)));
206 }
207
208 inf::Orbitable::init_orbits(group_with_unknown);
209 } else {
210 inf::Orbitable::init_orbits(marginal_symmetries);
211 }
212}

214void inf::DualVector::init_quovec_index_maps() {
215 ASSERT_TRUE(orbits_initialized())
216 ASSERT_EQUAL(m_quovec_index_to_orbit.size(), 0)
217 ASSERT_EQUAL(m_event_to_quovec_index.size(), 0)
218 ASSERT_EQUAL(m_n_orbits_no_unknown, 0)
219 ASSERT_EQUAL(m_orbit_repr_no_unknown.size(), 0)
220
221 // Init storage
222 m_quovec_index_to_orbit = std::vector<std::vector<inf::EventTensor::EventHash>>(
223 get_n_orbits(),
224 std::vector<inf::EventTensor::EventHash>());
225
226 m_event_to_quovec_index = std::vector<inf::QuovecIndex>(get_n_events());
227
228 inf::QuovecIndex quovec_index = 0;
229
230 for (bool store_unknowns : std::vector<bool>({false, true})) {
231 if (store_unknowns)
232 m_n_orbits_no_unknown = quovec_index;
233
234 for (inf::Orbitable::OrbitPair const &orbit_pair : get_orbits()) {
235 bool contains_unknown = false;
236 for (inf::Outcome outcome : orbit_pair.first) {
237 if (outcome == m_inflation->get_network()->get_outcome_unknown()) {
238 contains_unknown = true;
239 break;
240 }
241 }
242
243 // The first pass only stores the orbits with no unknowns, the second round
244 // stores only the orbits with unknowns
245 if (contains_unknown) {
246 if (not store_unknowns)
247 continue;
248 } else {
249 if (store_unknowns)
250 continue;
251
252 m_orbit_repr_no_unknown.push_back(orbit_pair.first);
253 }
254
255 for (inf::Event const &event : orbit_pair.second) {
256 inf::EventTensor::EventHash const event_hash = m_event_tensor.get_event_hash(event);
257
258 // Append the hashes of the orbit
259 m_quovec_index_to_orbit[quovec_index].push_back(event_hash);
260
261 // Mark the event as pointing to the current orbit
262 m_event_to_quovec_index[event_hash] = quovec_index;
263 }
264
265 ++quovec_index;
266 }
267 }
268
269 ASSERT_EQUAL(quovec_index, get_n_orbits());
270}

281void inf::DualVector::init_bound_rules() {
282 if (m_store_bounds == inf::DualVector::StoreBounds::no)
283 return;
284
285 inf::Outcome const unknown_outcome = m_inflation->get_network()->get_outcome_unknown();
286
287 for (inf::Orbitable::OrbitPair const &orbit_pair : get_orbits()) {
288 inf::Event const &repr_event = orbit_pair.first;
289
290 // Ensure that repr_event has at least one unknown
291 Index const current_n_unknowns = std::count(repr_event.begin(), repr_event.end(), unknown_outcome);
292 if (current_n_unknowns == 0)
293 continue;
294
295 // Now can append to the update rules
296 inf::QuovecIndex const to_update = m_event_to_quovec_index[get_event_tensor().get_event_hash(repr_event)];
297
298 inf::Event other_event = repr_event;
299 inf::Event::iterator const outcome_to_modify = std::find(other_event.begin(), other_event.end(), unknown_outcome);
300
301 // This is essentially the orbit of repr_event, but stored as quovec indices
302 std::set<inf::QuovecIndex> feasible_set;
303
304 for (inf::Outcome const other_outcome : util::Range(m_inflation->get_network()->get_n_outcomes())) {
305 *outcome_to_modify = other_outcome;
306
307 feasible_set.insert(m_event_to_quovec_index[get_event_tensor().get_event_hash(other_event)]);
308 }
309
310 std::vector<inf::QuovecIndex> const feasible_vec(feasible_set.begin(), feasible_set.end());
311
312 m_bound_rules.push_back(inf::DualVector::BoundRule(to_update, feasible_vec));
313 }
