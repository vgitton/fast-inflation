74bool inf::TreeSplitter::try_to_split_at_depth(Index const depth) {
75
76 Index const breadth = m_event_tree.get_breadth_at_depth(depth);
77
78 // Need the breadth to be at least as large as the number of splits to have
79 // a non-empty set of paths to look at by each thread
80 if (breadth < m_n_splits)
81 return false;
82
83 std::vector<PathAndLeaves> paths_and_leaves = get_paths_and_leaves(depth, breadth);
84 ASSERT_EQUAL(paths_and_leaves.size(), breadth)
85
86 // This list contains {0, i1, i2, i3, breadth}
87 // where the first thread looks at [0,i1), the second at [i1,i2), etc.
88 std::vector<Index> splits;
89 if (not find_splits(splits, paths_and_leaves)) {
90 // The splits could not be found at this depth, e.g., because the number of leaves of the paths
91 // at this depth look like (1,1,1,...,1,very large number), and so the first subtree took all
92 // paths for itself, which doesn't make sense
93 return false;
94 }
95 // splits now contain the best splits that we could find at this depth.
96 // Let's check how good this splitting is.
97 // The number of leaves per split is the metric to check that the split was fairly even.
98 std::vector<Index> leaves_per_split(m_n_splits, 0);
99 for (Index split_index : util::Range(m_n_splits)) {
100 for (Index path_index : util::Range(splits[split_index],
101 splits[split_index + 1])) {
102 leaves_per_split[split_index] += paths_and_leaves[path_index].n_leaves;
103 }
104 }
105 // Compute the quality factor
106 {
107 auto const min_max = std::minmax_element(leaves_per_split.begin(), leaves_per_split.end());
108 m_quality_factor = static_cast<double>(*min_max.second) / static_cast<double>(*min_max.first);
109 ASSERT_LTE(1.0, m_quality_factor)
110 }
111 // The success of the split depends on how low the quality factor is
112 if (m_quality_factor > inf::TreeSplitter::target_quality_factor) {
113 // The quality factor is not good enough
114 return false;
115 }
116
117 // Display
118 for (Index i : util::Range(m_n_splits)) {
119 util::logger
120 << "Thread " << i + 1
121 << " gets the range [" << splits[i] << "," << splits[i + 1] << ") "
122 << "with " << leaves_per_split[i] << " leaves" << util::cr;
123 }
124
125 // We have found a good split! We can set m_path_partition accordingly.
126 m_path_partition = std::make_shared<PathPartition>(m_n_splits);
127
128 for (Index split_index : util::Range(m_n_splits)) {
129 ASSERT_LT(splits[split_index], splits[split_index + 1])
130 Index const split_size = static_cast<Index>(splits[split_index + 1] - splits[split_index]);
131 (*m_path_partition)[split_index] = std::vector<Path>(split_size);
132
133 for (Index path_index : util::Range(split_size)) {
134 (*m_path_partition)[split_index][path_index] = paths_and_leaves[splits[split_index] + path_index].path;
135 }
136 }
137
138 return true;
139}

154 inf::EventTree::NodePos::Queue queue = m_event_tree.get_root_children_queue();

157
158 // This method does a bit too many calculations, I guess, but it works

187bool inf::TreeSplitter::find_splits(std::vector<Index> &splits,
188 std::vector<PathAndLeaves> const &paths_and_leaves) const {
189
190 Index const breadth = paths_and_leaves.size();
191
192 // This list contains {0, i1, i2, i3, breadth}
193 // where the first thread looks at [0,i1), the second at [i1,i2), etc.
194 splits = std::vector<Index>(m_n_splits + 1);
195 splits[0] = 0;
196 splits.back() = breadth;
197
198 Index accumulated_leaves = 0;
199 Index split_index = 1;
200
201 for (Index path_index : util::Range(breadth)) {
202
203 accumulated_leaves += paths_and_leaves[path_index].n_leaves;
204
205 if (accumulated_leaves > m_leave_thresholds[split_index]) {
206 // We cannot have i1 = 0, this would mean the first thread gets no leaves!
207 if (path_index == 0)
208 return false;
209
210 splits[split_index] = path_index;
211 ++split_index;
212
213 // If we found all our splits, we exit. This condition is necessary, for
214 // otherwise we'd be trying to set up splits.back(), which doesn't make sense
215 if (split_index == splits.size() - 1)
216 break;
217 }
218 }
219
220 // Normally, we should have set all the way up to i3, leaving split_index to 4.
221 // If not, this means that the accumulated leaves could look for instance like {1,1,1,1,1,1,1,(large number)}.
222 // Then i1 = 7 and we exit the above loop without having set the other i's.
223 return split_index == splits.size() - 1;
224}
