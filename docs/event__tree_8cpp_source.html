432void inf::EventTree::init_info() {
433 // sizeof(inf::EventTree) accounts for the std::vector m_node_cache and the
434 // root index
435 m_memory_footprint = sizeof(inf::EventTree) +
436 m_depth * sizeof(std::vector<inf::EventTree::Node>);
437
438 inf::EventTree::NodePos::Queue queue = get_root_children_queue();
439
440 // Number of leaves
441 m_n_leaves = 0;
442 for (inf::EventTree::NodePos const &root_node_pos : queue) {
443 m_n_leaves += count_leaves_from(root_node_pos);
444 }
445
446 // Number of nodes
447 m_n_nodes_total = 0;
448 while (not queue.empty()) {
449 inf::EventTree::NodePos node_pos = util::pop_back(queue);
450
451 ++m_n_nodes_total;
452
453 if (node_pos.depth < get_depth() - 1) {
454 add_children_to_queue(queue, node_pos);
455 }
456 }
457
458 for (Index const depth : util::Range(m_depth)) {
459 for (inf::EventTree::Node const &node : m_node_cache[depth]) {
460 m_memory_footprint += node.get_memory_footprint();
461 }
462 }
