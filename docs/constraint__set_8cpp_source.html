171double simplify_by_gcd(std::vector<mpz_t> &scale_factors) {
172 mpz_t one;
173 mpz_init_set_si(one, 1);
174
175 mpz_t the_gcd;
176 mpz_init_set(the_gcd, scale_factors[0]);
177
178 for (Index const i : util::Range(Index(1), scale_factors.size())) {
179 mpz_gcd(the_gcd, the_gcd, scale_factors[i]);
180 if (mpz_cmp(the_gcd, one) == 0)
181 return 1.0;
182 }
183
184 for (mpz_t &scale_factor : scale_factors) {
185 // Floor division
186 mpz_fdiv_q(scale_factor, scale_factor, the_gcd);
187 }
188
189 return mpz_get_d(the_gcd);
190}

211void inf::ConstraintSet::update_constraint_scale_factors() {
212 ASSERT_LT(0, m_constraints.size())
213
214 // One scale factor for the LHS and the RHS of each constraint
215 std::vector<mpz_t> scale_factors(2 * m_constraints.size());
216 for (mpz_t &scale_factor : scale_factors) {
217 mpz_init_set_si(scale_factor, 1);
218 }
219
220 // We want
221 // scale_factors[0] = RHS[0]*LHS[1]*RHS[1]*...
222 // scale_factors[1] = LHS[0] *LHS[1]*RHS[1]*...
223 // scale_factors[2] = LHS[0]*RHS[0] *RHS[1]*...
224 // scale_factors[3] = LHS[0]*RHS[0]*LHS[1] *...
225
226 m_quovec_denom = 1.0;
227
228 mpz_t lhs_denom, rhs_denom;
229 mpz_init(lhs_denom);
230 mpz_init(rhs_denom);
231
232 for (Index const constraint_index : util::Range(m_constraints.size())) {
233 m_quovec_denom *=
234 static_cast<double>(m_constraints[constraint_index]->get_lhs_denom()) * static_cast<double>(m_constraints[constraint_index]->get_rhs_denom());
235
236 signed long int lhs_denom = util::safe_ll_to_l(m_constraints[constraint_index]->get_lhs_denom());
237 signed long int rhs_denom = util::safe_ll_to_l(m_constraints[constraint_index]->get_rhs_denom());
238
239 for (Index const constraint_index_bis : util::Range(m_constraints.size())) {
240 mpz_t *lhs_scale_factor = &scale_factors[2 * constraint_index_bis];
241 mpz_t *rhs_scale_factor = &scale_factors[2 * constraint_index_bis + 1];
242
243 mpz_mul_si(*lhs_scale_factor, *lhs_scale_factor, rhs_denom);
244 mpz_mul_si(*rhs_scale_factor, *rhs_scale_factor, lhs_denom);
245
246 if (constraint_index == constraint_index_bis)
247 continue;
248
249 mpz_mul_si(*lhs_scale_factor, *lhs_scale_factor, lhs_denom);
250 mpz_mul_si(*rhs_scale_factor, *rhs_scale_factor, rhs_denom);
251 }
252 }
253
254 double const the_gcd = util::simplify_by_gcd(scale_factors);
255 m_quovec_denom /= the_gcd;
256
257 for (Index const constraint_index : util::Range(m_constraints.size())) {
258 inf::Constraint &constraint = *m_constraints[constraint_index];
259 constraint.set_lhs_scale(
260 util::mpz_to_ll(scale_factors[2 * constraint_index]));
261 constraint.set_rhs_scale(
262 util::mpz_to_ll(scale_factors[2 * constraint_index + 1]));
263 }
264
265 // Max dual_vector component
266
267 // m_max_dual_vector_component = MAX * gcd / (LHS[0]*RHS[0]*... * 2 * #constraints )
268 m_max_dual_vector_component = std::numeric_limits<Num>::max();
269 // Just to be safe
270 Num const safety_factor = 10;
271
272 Num const divide_by = safety_factor * util::mpz_to_ll(scale_factors[0]) * m_constraints[0]->get_lhs_denom() * 2 * static_cast<Num>(m_constraints.size());
273
274 m_max_dual_vector_component /= divide_by;
275
276 // To ensure that the quovecs always have small enough values, we can reset it here
277 {
278 inf::Quovec const zero_quovec(get_quovec_size(), 0);
279 set_dual_vector_from_quovec(zero_quovec);
280 }
281}
